SQL UPIT ZA INVENTAR

SELECT donor.KrvnaGrupa, sum(donor.KolicinaKrvi) as KolicinaKrvi
from donor left join inventar on inventar.donor=donor.idDonor
WHERE inventar.Donirano is null
group by donor.KrvnaGrupa
